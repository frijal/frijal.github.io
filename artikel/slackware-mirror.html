#!/bin/bash
#slackware-mirror.sh‚Äîpilihmirrorslackpkgotomatis

set-e

MIRROR_FILE="/etc/slackpkg/mirrors"
BACKUP="/etc/slackpkg/mirrors.backup"

sudocp"$MIRROR_FILE""$BACKUP"

echo"[INFO]Menampilkandaftarmirror(resmiSlackware):"

echo""

echo"[ERROR]Tidakadamirroryangdipilih.Keluar."
exit1
fi

echo"[INFO]Mengaturmirrorke$MIRROR_URL..."
echo"$MIRROR_URL"|sudotee"$MIRROR_FILE">/dev/null

sudoslackpkgupdate

echo"[DONE]Mirrorslackpkgberhasildiaturke$MIRROR_URL"
