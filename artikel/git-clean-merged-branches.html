set-e

GREEN="\033[1;32m"
YELLOW="\033[1;33m"
RED="\033[1;31m"
CYAN="\033[1;36m"
NC="\033[0m"

if!gitrev-parse--is-inside-work-tree&>/dev/null;then

exit1
fi

AUTO=false
DRY_RUN=false

#Bacaargumen
forargin"$@";do
case$argin
--auto)AUTO=true;;
--dry-run)DRY_RUN=true;;
esac
done

if[-z"$MERGED_BRANCHES"];then
echo-e"${GREEN}âœ…Tidakadabranchyangsudahdi-merge.${NC}"
exit0
fi

echo-e"\n${YELLOW}ğŸ§ªModedry-runaktif:tidakadabranchyangdihapus.${NC}"
exit0
fi

if!$AUTO;then

[[!"$confirm"=~^[Yy]$]]&&echo-e"${RED}ğŸš«Dibatalkan.${NC}"&&exit0
fi

echo-e"${YELLOW}ğŸ§¹Menghapusbranchlokal...${NC}"
whileread-rbranch;do
[-z"$branch"]&&continue
gitbranch-d"$branch"||echo-e"${RED}âŒGagalhapus$branch${NC}"
done<<<"$MERGED_BRANCHES"

if$AUTO;then
DELETE_REMOTE=true
