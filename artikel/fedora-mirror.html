fedora-mirror.sh—AktifkanMirrorTercepat(Fedora)

Skripfedora-mirror.sh:AktifkanMirrorTercepat

Fedoramenggunakandnfsebagaimanajerpaket.Untukmempercepatupdatedaninstalasi,kitabisamengaktifkanpluginfastestmirroryangakanotomatismemilihservermirrortercepat.SkripberikutakanmelakukannyauntukAnda.

⚠️Jalankandengansudokarenaskripiniakanmengubahfilekonfigurasisistemdi/etc/dnf/dnf.conf.

KodeSkrip

#!/bin/bash
#fedora-mirror.sh-MengaktifkanmirrorDNFtercepatdiFedora

set-e

echo"[INFO]Mem-backupkonfigurasiDNF..."
sudocp/etc/dnf/dnf.conf/etc/dnf/dnf.conf.bak

echo"[INFO]Mengaktifkan'fastestmirror'di/etc/dnf/dnf.conf..."
if!grep-q"fastestmirror=True"/etc/dnf/dnf.conf;then
echo"fastestmirror=True"|sudotee-a/etc/dnf/dnf.conf
fi

echo"[INFO]MembersihkancacheDNF..."
sudodnfcleanall

echo"[SELESAI]Plugin'fastestmirror'berhasildiaktifkan!"Copy
